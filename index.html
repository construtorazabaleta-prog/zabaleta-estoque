function doGet() { return json({ ok: true, msg: 'Estoque Tower Web App v2 ativo' }); } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const email = verificarToken(body.idToken); Utilities.formatDate(new Date(d.data), Session.getScriptTimeZone(), "yyyy-MM-dd'T'HH:mm:ss") : agora; linha = [ dt, d.id || '', d.numero || '', d.nome || '', d.origem || '', d.evento || '', d.detalhe || '', d.status || '', email, d.estado || '' ]; } else if (aba === 'ENTRADAS') { // Lê o cabeçalho da ENTRADAS (layout NF-e: 2 blocos de Quantidade/Unid./Preço) // e posiciona os valores pelo NOME da coluna — robusto à posição. var headRow = sheet.getRange(1, 1, 1, Math.max(1, sheet.getLastColumn())).getValues()[0]; var row = headRow.map(function(){ return ''; }); var q = num(d.quantidade), p = num(d.preco); var hist = (d.origem || '') + (d.origemDetalhe ? ' - ' + d.origemDetalhe : ''); for (var i = 0; i < headRow.length; i++) { var t = String(headRow[i]).trim(); if (/^data/i.test(t)) row[i] = agora; else if (/movimento/i.test(t)) row[i] = 'Entrada'; else if (/insumo/i.test(t)) row[i] = d.insumo || ''; else if (/forn/i.test(t)) row[i] = d.fornecedor || ''; else if (/hist/i.test(t)) row[i] = hist; else if (/^quantidade/i.test(t)) row[i] = q; // preenche os dois blocos else if (/^unid/i.test(t)) row[i] = d.unid || ''; else if (/pre[cç]o/i.test(t)) row[i] = p; else if (/total/i.test(t)) row[i] = (q !== '' && p !== '') ?
